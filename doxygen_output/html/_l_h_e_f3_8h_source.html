22 // The XMLTag struct is used to represent all information within an XML tag.
23 // It contains the attributes as a map, any sub-tags as a vector of pointers
24 // to other XMLTag objects, and any other information as a single string.
25 // The XMLTag struct written by Leif Lonnblad.
26 
27 struct XMLTag {
28 
29  // Convenient typdef.
30  typedef string::size_type pos_t;
31 
32  // Convenient alias for npos.
33  static const pos_t end;
34 
35  // The destructor also destroys any sub-tags.
36  ~XMLTag() {
37  for ( int i = 0, N = tags.size(); i < N; ++i )
38  if (tags[i]) delete tags[i];
39  }
40 
41  // The name of this tag.
42  string name;
43 
44  // The attributes of this tag.
45  map<string,string> attr;
46 
47  // A vector of sub-tags.
48  vector<XMLTag*> tags;
49 
50  // The contents of this tag.
51  string contents;
52 
53  // Find an attribute named n and set the double variable v to
54  // the corresponding value. Return false if no attribute was found.
55  bool getattr(string n, double & v) const {
56  map<string,string>::const_iterator it = attr.find(n);
57  if ( it == attr.end() ) return false;
58  v = atof(it->second.c_str());
59  return true;
60  }
61 
62  // Find an attribute named n and set the bool variable v to true if the
63  // corresponding value is "yes". Return false if no attribute was found.
64  bool getattr(string n, bool & v) const {
65  map<string,string>::const_iterator it = attr.find(n);
66  if ( it == attr.end() ) return false;
67  if ( it->second == "yes" ) v = true;
68  return true;
69  }
70

72  // corresponding value. Return false if no attribute was found.

81  // corresponding value. Return false if no attribute was found.

83  map<string,string>::const_iterator it = attr.find(n);

86  return true;
87  }
88 
89  // Find an attribute named n and set the string variable v to the
90  // corresponding value. Return false if no attribute was found.
91  bool getattr(string n, string & v) const {
92  map<string,string>::const_iterator it = attr.find(n);
93  if ( it == attr.end() ) return false;

98  // Scan the given string and return all XML tags found as a vector
99  // of pointers to XMLTag objects.
100  static vector<XMLTag*> findXMLTags(string str,
101  string * leftover = 0) {
102  vector<XMLTag*> tags;
103  pos_t curr = 0;
104 
105  while ( curr != end ) {
106 
107  // Find the first tag.
108  pos_t begin = str.find("<", curr);
109 
110  // Skip comments.
111  if ( begin != end && str.find("<!--", curr) == begin ) {
112  pos_t endcom = str.find("-->", begin);
113  if ( endcom == end ) {
114  if ( leftover ) *leftover += str.substr(curr);
115  return tags;
116  }
117  if ( leftover ) *leftover += str.substr(curr, endcom - curr);
118  curr = endcom;
119  continue;
120  }
121 
122  // Do not extract tags from #-commented lines.
123  string id = str.substr(begin + 1, str.find_first_of(" \t\n/>",
124  begin)-begin-1);
125  while(id.find(" ", 0) != string::npos)
126  id.erase(id.begin()+id.find(" ",0));
127  pos_t lasthash = str.find_last_of("#", begin);
128  pos_t lastbreak = str.find_last_of("\n", begin);
129  pos_t now = str.find("<"+id, curr);
130  if ( begin != end
131  && lasthash != string::npos
132  && (lastbreak != string::npos || curr == 0)
133  && (lasthash > lastbreak || curr == 0)
134  && now != string::npos
135  && lasthash < now) {
136  pos_t endcom = str.find("</"+id+">", begin);
137  pos_t endcom2 = str.find_first_of("\n", begin);
138  if ( endcom == end && endcom2 == end) {
139  if ( leftover ) *leftover += str.substr(curr);
140  return tags;
141  } else if ( endcom == end)
142  endcom = endcom2;
143  else
144  endcom += ("</"+id+">").size();
145  if ( leftover ) *leftover += str.substr(curr, endcom - curr);
146  curr = endcom;
147  continue;
148  }
149 
150  // Also skip CDATA statements.
151  // Used for text data that should not be parsed by the XML parser.
152  // (e.g., JavaScript code contains a lot of "<" or "&" characters
153  // which XML would erroneously interpret as the start of a new
154  // element or the start of a character entity, respectively.)
155  // See eg http://www.w3schools.com/xml/xml_cdata.asp
156  if ( str.find("<![CDATA[", curr) == begin ) {
157  pos_t endcom = str.find("]]>", begin);
158  if ( endcom == end ) {
159  if ( leftover ) *leftover += str.substr(curr);
160  return tags;
161  }
162  if ( leftover ) *leftover += str.substr(curr, endcom - curr);
163  curr = endcom;
164  continue;
165  }
166 
167  if ( leftover ) *leftover += str.substr(curr, begin - curr);
168  if ( begin == end || begin > str.length() - 3 || str[begin + 1] == '/' )
169  return tags;
170 
171  pos_t close = str.find(">", curr);
172  if ( close == end ) return tags;
173 
174  // Find the tag name.
175  curr = str.find_first_of(" \t\n/>", begin);
176  tags.push_back(new XMLTag());
177  tags.back()->name = str.substr(begin + 1, curr - begin - 1);
178 
179  while ( true ) {
180 
181  // Now skip some white space to see if we can find an attribute.
182  curr = str.find_first_not_of(" \t\n", curr);
183  if ( curr == end || curr >= close ) break;
184 
185  pos_t tend = str.find_first_of("= \t\n", curr);
186  if ( tend == end || tend >= close ) break;
187 
188  string name = str.substr(curr, tend - curr);
189  curr = str.find("=", curr) + 1;
190 
191  // OK now find the beginning and end of the atribute.
192  curr = str.find("\"", curr);
193  if ( curr == end || curr >= close ) break;
194  pos_t bega = ++curr;
195  curr = str.find("\"", curr);
196  while ( curr != end && str[curr - 1] == '\\' )
197  curr = str.find("\"", curr + 1);
198 
199  string value = str.substr(bega, curr == end? end: curr - bega);
200 
201  tags.back()->attr[name] = value;
202 
203  ++curr;
204 
205  }
206 
207  curr = close + 1;
208  if ( str[close - 1] == '/' ) continue;
209 
210  pos_t endtag = str.find("</" + tags.back()->name + ">", curr);
211  if ( endtag == end ) {
212  tags.back()->contents = str.substr(curr);
213  curr = endtag;
214  } else {
215  tags.back()->contents = str.substr(curr, endtag - curr);
216  curr = endtag + tags.back()->name.length() + 3;
217  }
218 
219  string leftovers;
220  tags.back()->tags = findXMLTags(tags.back()->contents, &leftovers);
221  if ( leftovers.find_first_not_of(" \t\n") == end ) leftovers="";
222  tags.back()->contents = leftovers;
223 
224  }
225 
226  return tags;
227 
228  }
229 
230  // Print out this tag to a stream.
231  void list(ostream & os) const {
232  os << "<" << name;
233  for ( map<string,string>::const_iterator it = attr.begin();
234  it != attr.end(); ++it )
235  os << " " << it->first << "=\"" << it->second << "\"";
236  if ( contents.empty() && tags.empty() ) {
237  os << "/>" << endl;
238  return;
239  }
240  os << ">" << endl;
241  for ( int i = 0, N = tags.size(); i < N; ++i )
242  tags[i]->list(os);
243 
244  os << "````" << contents << "''''</" << name << ">" << endl;
245  }

579  IDBMUP = x.IDBMUP;
