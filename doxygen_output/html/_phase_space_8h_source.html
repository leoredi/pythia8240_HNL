291  // Standard methods to find t range of a 2 -> 2 process
292  // and to check whether a given t value is in that range.
293  pair<double,double> tRange( double sIn, double s1In, double s2In,
294  double s3In, double s4In) {
295  double lambda12 = pow2( sIn - s1In - s2In) - 4. * s1In * s2In;
296  double lambda34 = pow2( sIn - s3In - s4In) - 4. * s3In * s4In;
297  if (lambda12 < 0. || lambda34 < 0.) return make_pair( 0., 0.);
298  double tLow = -0.5 * (sIn - (s1In + s2In + s3In + s4In) + (s1In - s2In)
299  * (s3In - s4In) / sIn + sqrtpos(lambda12 * lambda34) / sIn);
300  double tUpp = ( (s3In - s1In) * (s4In - s2In) + (s1In + s4In - s2In - s3In)
301  * (s1In * s4In - s2In * s3In) / sIn ) / tLow;
302  return make_pair( tLow, tUpp); }
303  bool tInRange( double tIn, double sIn, double s1In, double s2In,
304  double s3In, double s4In) {
305  pair<double, double> tRng = tRange( sIn, s1In, s2In, s3In, s4In);
306  return (tIn > tRng.first && tIn < tRng.second); }
